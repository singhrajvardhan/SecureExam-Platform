-- Database: exam_system

-- Students table (enhanced)
CREATE TABLE IF NOT EXISTS `students` (
    `id` INT AUTO_INCREMENT PRIMARY KEY,
    `application_id` VARCHAR(50) UNIQUE NOT NULL,
    `roll_number` VARCHAR(50) UNIQUE NOT NULL,
    `full_name` VARCHAR(100) NOT NULL,
    `email` VARCHAR(100) UNIQUE NOT NULL,
    `phone` VARCHAR(20),
    `password` VARCHAR(255) NOT NULL,
    `date_of_birth` DATE,
    `profile_photo` VARCHAR(255),
    `register_date` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Teachers table
CREATE TABLE IF NOT EXISTS `teachers` (
    `id` INT AUTO_INCREMENT PRIMARY KEY,
    `username` VARCHAR(50) UNIQUE NOT NULL,
    `email` VARCHAR(100) UNIQUE NOT NULL,
    `password` VARCHAR(255) NOT NULL,
    `full_name` VARCHAR(100) NOT NULL,
    `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Questions table
CREATE TABLE IF NOT EXISTS `questions` (
    `id` INT AUTO_INCREMENT PRIMARY KEY,
    `question_text` TEXT NOT NULL,
    `option_a` VARCHAR(255) NOT NULL,
    `option_b` VARCHAR(255) NOT NULL,
    `option_c` VARCHAR(255) NOT NULL,
    `option_d` VARCHAR(255) NOT NULL,
    `correct_answer` CHAR(1) NOT NULL,
    `marks` INT DEFAULT 4,
    `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- Exam sessions table (tracks each exam attempt)
CREATE TABLE IF NOT EXISTS `exam_sessions` (
    `id` INT AUTO_INCREMENT PRIMARY KEY,
    `student_id` VARCHAR(50) NOT NULL,
    `student_name` VARCHAR(100),
    `roll_number` VARCHAR(50),
    `start_time` DATETIME,
    `end_time` DATETIME,
    `status` ENUM('active', 'completed', 'terminated') DEFAULT 'active',
    `total_tab_switches` INT DEFAULT 0,
    `suspicious_activities` INT DEFAULT 0,
    `face_detection_fails` INT DEFAULT 0,
    `created_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`student_id`) REFERENCES `students`(`application_id`)
);

-- Exam answers table
CREATE TABLE IF NOT EXISTS `exam_answers` (
    `id` INT AUTO_INCREMENT PRIMARY KEY,
    `session_id` INT NOT NULL,
    `student_id` VARCHAR(50) NOT NULL,
    `question_id` INT NOT NULL,
    `selected_answer` CHAR(1),
    `is_correct` BOOLEAN DEFAULT FALSE,
    `answered_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`session_id`) REFERENCES `exam_sessions`(`id`),
    FOREIGN KEY (`question_id`) REFERENCES `questions`(`id`)
);

-- Tab switch logs
CREATE TABLE IF NOT EXISTS `tab_switch_logs` (
    `id` INT AUTO_INCREMENT PRIMARY KEY,
    `session_id` INT NOT NULL,
    `student_id` VARCHAR(50) NOT NULL,
    `switch_time` DATETIME,
    `switch_count` INT DEFAULT 1,
    `recorded_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`session_id`) REFERENCES `exam_sessions`(`id`)
);

-- Camera logs (face detection)
CREATE TABLE IF NOT EXISTS `camera_logs` (
    `id` INT AUTO_INCREMENT PRIMARY KEY,
    `session_id` INT NOT NULL,
    `student_id` VARCHAR(50) NOT NULL,
    `image_path` VARCHAR(255),
    `face_detected` BOOLEAN,
    `multiple_faces` BOOLEAN DEFAULT FALSE,
    `no_face` BOOLEAN DEFAULT FALSE,
    `detection_time` DATETIME,
    `recorded_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`session_id`) REFERENCES `exam_sessions`(`id`)
);

-- Exam results table
CREATE TABLE IF NOT EXISTS `exam_results` (
    `id` INT AUTO_INCREMENT PRIMARY KEY,
    `session_id` INT NOT NULL,
    `student_id` VARCHAR(50) NOT NULL,
    `roll_number` VARCHAR(50),
    `student_name` VARCHAR(100),
    `total_questions` INT,
    `correct_answers` INT,
    `wrong_answers` INT,
    `total_marks` INT,
    `obtained_marks` INT,
    `score_percentage` DECIMAL(5,2),
    `total_tab_switches` INT DEFAULT 0,
    `suspicious_activities` INT DEFAULT 0,
    `completed_at` TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (`session_id`) REFERENCES `exam_sessions`(`id`)
);
